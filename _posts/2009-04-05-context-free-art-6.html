Sometimes it snows in April


startshape main 
background{b -.2}
tile{s 3}

rule main {
  10 * {} Snows {}
}
 
rule Snows {
  SNOW {}
  Snows {a -.1 s .9 y 2 r 20}
}
rule Snows {
  SNOW {}
  Snows {a -.1 s .9 x 2 r -5}
}

rule SNOW {
  TRIANGLE {}
  SNOW {r 60 x .001 s .99 b .02}
}
